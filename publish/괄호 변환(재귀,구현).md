# 특정 문자열에서 균형잡힌 문자열의 인덱스 반환하는 함수
# 특정한 균형 문자열이 올바른 관호 문자열인지 판단 함수 
# 이후 재귀 함수에서 두 함수를 불러오도록 

# 균형잡힌 괄호 문자열의 인덱스 반환 
def balanced_index(p) :
    count = 0
    for i in range(len(p)):
        if p[i] == '(' :
            count += 1
        else :
            count -= 1
        if count == 0 :
            return i

def check_proper(p) : 
    count = 0
    for i in p :
        if i == '(' :
            count += 1
        else :
            if count == 0:
                return False
            count -= 1
    return True


def solution(p):
    answer = ''
    if p == '' :
        return answer
    index = balanced_index(p)
    u = p[:index + 1]
    v = p[index + 1:]
    if check_proper(u) :
        answer = u + solution(v)
    else :
        answer = '('
        answer += solution(v)
        answer += ')'
        u =list(u[1:-1])
        for i in range(len(u)) :
            if u[i] == '(' :
                u[i] = ')'
            else :
                u[i] = '('
        answer += ''.join(u)
    return answer

print(solution("()))((()"))
